## 快速读

+ 不支持负数

```cpp
const int BUF_SZ = 100000;
char buf[BUF_SZ + 10];

inline char nc() {
    static char *pr = buf, *pend = buf;
    if (pr == pend) {
        pr = buf;
        pend = pr + fread(buf, 1, BUF_SZ, stdin);
        if (pr == pend) return EOF;
        else return *pr++;
    }
    return *pr++;
}

template <typename T>
inline bool read(T& x) {
    static char ch;
    ch = nc();
    while (ch != EOF && !isdigit(ch)) ch = nc();
    if (ch == EOF) return 0;
    for (x = 0; isdigit(ch); ch = nc())
        x = x * 10 + ch - '0';
    return 1;
}
```


+ 需要初始化
+ 需要一次读入
+ 不支持负数

```cpp
const int MAXS = 100 * 1024 * 1024;
char buf[MAXS];
template<typename T>
inline bool read(T& x) {
    static char* p = buf;
    x = 0;
    while (*p && !isdigit(*p)) ++p;
    if (!*p) return false;
    while (isdigit(*p)) x = x * 10 + *p++ - 48;
    return true;
}

fread(buf, 1, MAXS, stdin);
```

## 对拍

```bash
#!/usr/bin/env bash
g++ -o r main.cpp -O2 -std=c++11
g++ -o std std.cpp -O2 -std=c++11
while true; do
    python gen.py > in
    ./std < in > stdout
    ./r < in > out
    if test $? -ne 0; then
        exit 0
    fi
    if diff stdout out; then
        printf "AC\n"
    else
        printf "GG\n"
        exit 0
    fi
done
```
